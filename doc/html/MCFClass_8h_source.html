02176

02178            FONumber( MCFQCoef( i ) ) * FONumber( tX[ i ] ) / 2 )

02191

02204  MCFGetPi( tPi );

02206  FONumber BY = 0;
02207  for( Index i = 0 ; i < MCFn() ; i++ )
02208   BY += FONumber( tPi[ i ] ) * FONumber( MCFDfct( i ) );

02211  MCFGetRC( tRC );

02214  MCFGetX( tX );
02215 
02216  #if( ! USENAME0 )
02217   tPi--;
02218  #endif
02219 
02220  FONumber QdrtcCmpnt = 0;  // the quadratic part of the objective
02221  for( Index i = 0 ; i < MCFm() ; i++ ) {
02222   if( IsClosedArc( i ) )

02224 
02225   cFONumber tXi = FONumber( tX[ i ] );
02226   cCNumber QCi = CNumber( MCFQCoef( i ) * tXi );

02239     throw( MCFException( "Csomplementary Slackness violated (CheckDSol)" ) );
02240    }
02241   else

02243     throw( MCFException( "Complementary Slackness violated (CheckDSol)" ) );
02244

02247  delete[] tX;
02248  delete[] tRC;
02249  #if( ! USENAME0 )
02250   tPi++;
02251  #endif
02252  delete[] tPi;
